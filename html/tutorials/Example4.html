from lamatrix import Spline

[5]:

knot_spacing = 0.05
order = 3
knots = np.arange(-5 - knot_spacing * (order - 1), 5 + (knot_spacing * (order - 1)) + knot_spacing, knot_spacing)

[6]:

model = Spline('x', knots=knots, order=order)

[7]:

model.fit(x=x, data=y)

[8]:
